```base
properties:
  note.property.Bezeichnung:
    displayName: Name
  file.name:
    displayName: File
views:
  - type: table
    name: Leichte Rüstungen
    filters:
      and:
        - tags.contains("Gegenstand/Rüstung/Leicht")
    order:
      - file.name
      - Kst.
      - RK
      - STR
      - Heiml.
      - Gew.
    sort:
      - property: file.name
        direction: ASC
      - property: property.Bezeichnung
        direction: ASC
    columnSize:
      file.name: 200
      note.Kst.: 75
      note.RK: 75
      note.STR: 50
      note.Heiml: 75
      note.Gew.: 60
      note.Eigenschaften: 150
```
